-- Return a string counting from 1 to the specified number.
-- Replace numbers dvisible by 3 with "fizz" and numbers divisible

-- replace it with "fizzbuzz".

fizzbuzz :: Int -> String
fizzbuzz x | x < 0 = error "Non-negative numbers only"
fizzbuzz 0 = ""
fizzbuzz 1 = "1"
fizzbuzz x = 
         if (mod x 15) == 0
